BluePrefab = nil
GreenPrefab = nil
RedPrefab = nil
WhitePrefab = nil
YellowPrefab = nil

Parent = nil

row = 8;
cloumn =10 

arrayIndex = {}

arrayObj = {}




function awake()
	print("lua awake...")
	Parent = CS.UnityEngine.GameObject.Find("Parent").transform

	BluePrefab  = CS.UnityEngine.Resources.Load("Prefabs/BlueSphere")
	--CS.UnityEngine.GameObject.Instantiate(BluePrefab,Parent)

	GreenPrefab  = CS.UnityEngine.Resources.Load("Prefabs/GreenSphere")
	--CS.UnityEngine.GameObject.Instantiate(GreenPrefab,Parent)

	RedPrefab  = CS.UnityEngine.Resources.Load("Prefabs/RedSphere")
	--CS.UnityEngine.GameObject.Instantiate(RedPrefab,Parent)

	WhitePrefab  = CS.UnityEngine.Resources.Load("Prefabs/WhiteSphere")
	--CS.UnityEngine.GameObject.Instantiate(WhitePrefab,Parent)

	YellowPrefab  = CS.UnityEngine.Resources.Load("Prefabs/YellowSphere")
	--CS.UnityEngine.GameObject.Instantiate(YellowPrefab,Parent)


end

function start()
	print("lua start...")
init()
end

function ondestroy()
    print("lua destroy")
end


function init()
 math.randomseed(os.time())
 math.random(1,10000)
for i=1,row do
	arrayIndex[i] = {}
		 for j = 1,cloumn do
			 arrayIndex[i][j] =  math.random(1,5)
		end
end

for i=1,row do
arrayObj[i] = {}
	  for j = 1,cloumn do
		--print(arrayIndex[i][j])
		tempGo = nil
		if(arrayIndex[i][j] == 1)
		then
			tempGo = CS.UnityEngine.GameObject.Instantiate(BluePrefab,Parent)
		elseif(arrayIndex[i][j] == 2)
		then
			tempGo = CS.UnityEngine.GameObject.Instantiate(GreenPrefab,Parent)
		elseif(arrayIndex[i][j] == 3)
		then
			tempGo = CS.UnityEngine.GameObject.Instantiate(RedPrefab,Parent)
		elseif(arrayIndex[i][j] == 4)
		then
			tempGo = CS.UnityEngine.GameObject.Instantiate(WhitePrefab,Parent)
		elseif(arrayIndex[i][j] == 5)
		then
			tempGo = CS.UnityEngine.GameObject.Instantiate(YellowPrefab,Parent)
		end
		tempGo.transform.localPosition = CS.UnityEngine.Vector3(i,j,10);
		arrayObj[i][j] = tempGo
	 end
end

end
